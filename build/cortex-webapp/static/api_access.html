curl -X GET "https://api.bharat-radar.vellur.in/api/v1/flights?airport=VIDP" -H "X-API-Key: br_your_key"
import requests
url = "https://api.bharat-radar.vellur.in/api/v1/flights"
headers = {"X-API-Key": "br_your_key"}
print(requests.get(url, headers=headers, params={"airport": "VIDP"}).json())
fetch('https://api.bharat-radar.vellur.in/api/v1/flights?airport=VIDP', 
  {headers: {'X-API-Key': 'br_your_key'}})
.then(r => r.json()).then(console.log)
